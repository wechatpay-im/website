```php [异步纯链式]
$instance->v3->vehicle->etc->transactions->id->_transaction_id_->getAsync([

  'query' => [
    'sub_mchid' => '1900000109',
  ],
])
->then(static function(\Psr\Http\Message\ResponseInterface $response) {
  print_r(json_decode((string) $response->getBody(), true));
})
->wait();
```
